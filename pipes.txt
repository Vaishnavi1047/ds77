//string 

#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <stdlib.h>

int main() {
    int p1[2], p2[2];
    char str1[100], str2[] = " Tiwari";
    char result[200];
    int i = 0, j = 0;

    if (pipe(p1) == -1 || pipe(p2) == -1) {
        perror("Pipe failed");
        return 1;
    }

    int pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        return 1;
    }

    if (pid > 0) {
        close(p1[0]);
        close(p2[1]);

        printf("Enter a string: ");
        fgets(str1, sizeof(str1), stdin);

        int k = 0;
        while (str1[k] != '\0') {
            if (str1[k] == '\n') {
                str1[k] = '\0';
                break;
            }
            k++;
        }

        write(p1[1], str1, strlen(str1) + 1);
        read(p2[0], result, sizeof(result));

        printf("Final string: %s\n", result);

        close(p1[1]);
        close(p2[0]);
    } 
    else {
        close(p1[1]);
        close(p2[0]);

        read(p1[0], str1, sizeof(str1));

        i = 0;
        while (str1[i] != '\0') {
            result[i] = str1[i];
            i++;
        }

        j = 0;
        while (str2[j] != '\0') {
            result[i] = str2[j];
            i++;
            j++;
        }

        result[i] = '\0';

        write(p2[1], result, strlen(result) + 1);

        close(p1[0]);
        close(p2[1]);
    }

    return 0;
}

//matrix
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>

#define R 2
#define C 2

int main() {
    int fd1[2], fd2[2];
    pid_t p;
    int mat1[R][C], mat2[R][C], sum[R][C];
    int i, j;

    if (pipe(fd1) == -1 || pipe(fd2) == -1) {
        return 1;
    }

    p = fork();

    if (p < 0) {
        return 1;
    } 
    else if (p > 0) {
        close(fd1[0]);
        close(fd2[1]);

        printf("Enter elements of Matrix 1 (2x2):\n");
        for (i = 0; i < R; i++)
            for (j = 0; j < C; j++)
                scanf("%d", &mat1[i][j]);

        printf("Enter elements of Matrix 2 (2x2):\n");
        for (i = 0; i < R; i++)
            for (j = 0; j < C; j++)
                scanf("%d", &mat2[i][j]);

        write(fd1[1], mat1, sizeof(mat1));
        write(fd1[1], mat2, sizeof(mat2));

        read(fd2[0], sum, sizeof(sum));

        printf("Sum of matrices:\n");
        for (i = 0; i < R; i++) {
            for (j = 0; j < C; j++) {
                printf("%d ", sum[i][j]);
            }
            printf("\n");
        }

        close(fd1[1]);
        close(fd2[0]);
    } 
    else {
        close(fd1[1]);
        close(fd2[0]);

        read(fd1[0], mat1, sizeof(mat1));
        read(fd1[0], mat2, sizeof(mat2));

        for (i = 0; i < R; i++) {
            for (j = 0; j < C; j++) {
                sum[i][j] = mat1[i][j] + mat2[i][j];
            }
        }

	//multiplication
        for (i = 0; i < R; i++) {
            for (j = 0; j < C; j++) {
                result[i][j] = 0;
                for (k = 0; k < C; k++) {
                    result[i][j] += mat1[i][k] * mat2[k][j];
                }
            }
        }

        write(fd2[1], sum, sizeof(sum));

        close(fd1[0]);
        close(fd2[1]);
    }

    return 0;
}
