// file having maximum size in the current working directory
#include <stdio.h>
#include <stdlib.h>
#include <dirent.h>
#include <sys/stat.h>
#include <pwd.h>
#include <string.h>

int main() {
    struct dirent *entry;
    DIR *dp = opendir(".");

    if (dp == NULL) {
        perror("opendir");
        return 1;
    }

    off_t max_size = -1;
    char max_user[256];

    while ((entry = readdir(dp))) {
        struct stat st;

        if (stat(entry->d_name, &st) == -1)
            continue;

        if (!S_ISREG(st.st_mode))
            continue;

        if (st.st_size > max_size) {
            max_size = st.st_size;

            struct passwd *pw = getpwuid(st.st_uid);
            strcpy(max_user, pw->pw_name);
        }
    }

    closedir(dp);

    printf("User owning the largest file: %s (Size: %lld bytes)\n",
           max_user, (long long)max_size);

    return 0;
}
