#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[]) {
    int rank, size;
    int numbers[2], result;

    MPI_Init(&argc, &argv);

    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    if (size < 2) {
        if (rank == 0)
            printf("Run with at least 2 processes\n");
        MPI_Finalize();
        return 0;
    }

    if (rank == 0) {
        numbers[0] = 10;
        numbers[1] = 20;

        printf("Client sending numbers: %d and %d\n",
               numbers[0], numbers[1]);

        MPI_Send(numbers, 2, MPI_INT, 1, 0, MPI_COMM_WORLD);

        MPI_Recv(&result, 1, MPI_INT, 1, 1,
                 MPI_COMM_WORLD, MPI_STATUS_IGNORE);

        printf("Client received result: %d\n", result);
    }

    if (rank == 1) {
        MPI_Recv(numbers, 2, MPI_INT, 0, 0,
                 MPI_COMM_WORLD, MPI_STATUS_IGNORE);

        result = numbers[0] + numbers[1];

        printf("Worker computed sum: %d\n", result);

        MPI_Send(&result, 1, MPI_INT, 0, 1, MPI_COMM_WORLD);
    }

    MPI_Finalize();
    return 0;
}

mpicc program.c -o program
mpirun -np 2 ./program


#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[]) {
    int rank, size;
    int number, result;

    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if (rank == 0) {
        number = 5;
        printf("Master sending number %d\n", number);

        for (int i = 1; i < size; i++) {
            MPI_Send(&number, 1, MPI_INT, i, 0, MPI_COMM_WORLD);
        }

        for (int i = 1; i < size; i++) {
            MPI_Recv(&result, 1, MPI_INT, i, 0,
                     MPI_COMM_WORLD, MPI_STATUS_IGNORE);
            printf("Received result %d from process %d\n", result, i);
        }
    }
    else {
        MPI_Recv(&number, 1, MPI_INT, 0, 0,
                 MPI_COMM_WORLD, MPI_STATUS_IGNORE);

        result = number * number;

        printf("Process %d computed: %d\n", rank, result);

        MPI_Send(&result, 1, MPI_INT, 0, 0, MPI_COMM_WORLD);
    }

    MPI_Finalize();
    return 0;
}
